QUERY 1:

SELECT ID,NAME FROM CUSTOMER ORDER BY NAME DESC, ID ASC


QUERY 2 :

 SELECT ID,NAME, MARKS, 
    CASE WHEN (MARKS) > 90
    THEN 'A+'
    WHEN MARKS BETWEEN 70 AND 90
    THEN 'A'
    WHEN MARKS BETWEEN 50 AND 70
    THEN 'B'
    WHEN MARKS BETWEEN 40 AND 50
    THEN 'C'
    ELSE 'FAIL'
    END AS GRADE
    FROM RESULTS STUDENTS;
